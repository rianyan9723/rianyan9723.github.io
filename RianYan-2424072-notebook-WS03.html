for cc in content:
    for c in cc: 
        count+= 1
         
        print('\nMovie review ', count)
        #Get review.
        str = c.find_all('a', attrs={'class':'title'}) 
        rReview =''

            rVal.append(str1)
            
        val = rVal[0]
        
        if(len(val) > 2): 
            continue
        else: 
            review.append(rReview) 
            rating.append(val)

            
            print('Review: ', rReview)
            print('Rating: ',val)

def textblob_tokenizer(input_str):
    blob = TextBlob(input_str.lower())
    tokens = blob.words
    words = [token.stem() for token in tokens] 
    return words
